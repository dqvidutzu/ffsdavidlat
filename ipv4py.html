            import requests

def ip_info(ip):
    url = f"https://ipwhois.app/json/{ip}"
    response = requests.get(url)
    
    if response.status_code == 200:
        ip_info = response.json()
        if ip_info.get("success", True) == False:
            print(f"error: {ip_info['message']}")
        else:
            print("--------------------------------")
            for key, value in ip_info.items():
                print(f"{key.lower()}: {value}")
            print("---------------------------------")
            
            print("Main: ")
            main = ['ip', 'country', 'city', 'continent']
            for th in main:
                if th in ip_info:
                    print(f"{th.lower()}: {ip_info[th]}")
    else:
        print("error: {ip_info['message']}")

ipaddress = input("IP: ")
ip_info(ipaddress)
